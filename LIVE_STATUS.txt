1. Clone repository
   cd /opt && sudo git clone <repo-url> mcp-lsp-bridge

2. Build
   npm install && npm run build

3. Deploy service
   sudo ./scripts/deploy.sh system

4. Setup Nginx + SSL
   sudo ./scripts/setup-nginx.sh qenex.ai
   sudo certbot --nginx -d qenex.ai

5. Verify
   ./scripts/verify-deployment.sh qenex.ai 3000 true

6. Done! Your MCP URL: https://qenex.ai/mcp
